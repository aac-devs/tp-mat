		\vec{b}=(2, 1, -1) \hspace*{1cm}

\end{center}
\begin{multicols}{2}

	\begin{align*}

	\end{align*}
	\noindent Calculamos $\vec{b'}$:
	\begin{align*}
		\vec{b'} & = \vec{b} - \vec{p}                                                           \\
		\vec{b'} & = (2, 1, -1) - \left( \cfrac{39}{38}, \cfrac{26}{38}, -\cfrac{65}{38} \right) \\
